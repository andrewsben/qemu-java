005package org.anarres.qemu.qapi.common;
006
007import com.google.gson.Gson;
008import com.google.gson.GsonBuilder;
009import java.io.Closeable;
010import java.io.IOException;
011import java.io.InputStreamReader;
012import java.io.OutputStreamWriter;
013import java.io.Reader;
014import java.io.Writer;
015import java.net.InetSocketAddress;
016import java.net.Socket;
017import java.nio.charset.Charset;
018import javax.annotation.CheckForNull;
019import javax.annotation.Nonnull;
020
021/**
022 * A connection to a QEmu process.
023 *
024 * @author shevek
025 */
026public class QApiConnection implements Closeable {
027
028    private static final Charset ISO_8859_1 = Charset.forName("ISO-8859-1");
029    private final Gson gson = new GsonBuilder().create();
030    private final Socket socket;
031    private final Writer output;
032    private final Reader input;
033    private final QApiGreeting greeting;
034
035    public QApiConnection(@Nonnull Socket socket) throws IOException {
036        this.socket = socket;
037        this.output = new OutputStreamWriter(socket.getOutputStream(), ISO_8859_1);
038        this.input = new InputStreamReader(socket.getInputStream(), ISO_8859_1);
039        this.greeting = gson.fromJson(input, QApiGreeting.class);
040        invoke(new QmpCapabilitiesCommand());
041    }
042
043    public QApiConnection(@Nonnull InetSocketAddress address) throws IOException {
044        this(new Socket(address.getAddress(), address.getPort()));
045    }
046
047    @Nonnull
048    public QApiGreeting getGreeting() {
049        return greeting;
050    }
051
052    @Nonnull
053    public QApiGreeting.QEmuVersion getQEmuVersion() {
054        return getGreeting().QMP.version.qemu;
055    }
056
057    @Nonnull
058    public <Argument, Response extends QApiResponse<?>> Response invoke(@Nonnull QApiCommand<Argument, Response> command) {
059        gson.toJson(command, output);
060        Class<Response> returnType = command.getReturnType();
061        return gson.fromJson(input, returnType);
062    }
063
064    @Override
065    public void close() throws IOException {
066        socket.close();
067    }
068}
