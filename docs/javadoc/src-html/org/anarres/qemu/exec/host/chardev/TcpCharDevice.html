005package org.anarres.qemu.exec.host.chardev;
006
007import java.net.InetSocketAddress;
008import java.util.Map;
009import javax.annotation.Nonnull;
010
011/**
012 *
013 * @author shevek
014 */
015public class TcpCharDevice extends AbstractSocketCharDevice {
016
017    private final InetSocketAddress address;
018    public boolean nodelay;
019
020    public TcpCharDevice(@Nonnull InetSocketAddress address, boolean server, boolean nowait) {
021        super("tcp", server, nowait);
022        this.address = address;
023        if (address.getHostString() == null && !server)
024            throw new IllegalArgumentException("A client tcp chardev requires a hostname.");
025    }
026
027    public TcpCharDevice(InetSocketAddress address) {
028        this(address, address.getHostString() == null, false);
029    }
030
031    public TcpCharDevice(int port) {
032        this(new InetSocketAddress(port), false, false);
033    }
034
035    @Nonnull
036    public InetSocketAddress getAddress() {
037        return address;
038    }
039
040    @Override
041    protected void addProperties(Map<String, Object> m) {
042        super.addProperties(m);
043        String host = address.getHostString();
044        if (host != null) {
045            if (server)
046                m.put("host", host);
047            else
048                m.put("to", host);
049        }
050        m.put("port", address.getPort());
051        if (nodelay)
052            m.put("nodelay", null);
053    }
054
055    @Override
056    public String toString() {
057        StringBuilder buf = new StringBuilder("tcp:");
058        buf.append(UdpCharDevice.toHostString(address))
059                .append(':').append(String.valueOf(address.getPort()));
060        if (server)
061            buf.append(",server");
062        if (nowait)
063            buf.append(",nowait");
064        if (nodelay)
065            buf.append(",nodelay");
066        return buf.toString();
067    }
068}
