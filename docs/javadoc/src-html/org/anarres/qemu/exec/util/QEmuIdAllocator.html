005package org.anarres.qemu.exec.util;
006
007import javax.annotation.Nonnegative;
008import javax.annotation.Nonnull;
009
010/**
011 *
012 * @author shevek
013 */
014public class QEmuIdAllocator {
015
016    private int driveIndex;
017    private int networkIndex;
018    private int pciAddress;
019
020    @Nonnegative
021    public int newDriveIndex() {
022        return ++driveIndex;
023    }
024
025    @Nonnegative
026    public int newNetworkIndex() {
027        return ++networkIndex;
028    }
029
030    @Nonnull
031    public String newPciAddresses(int n, String sep) {
032        StringBuilder buf = new StringBuilder();
033        for (int i = 0; i < n; i++) {
034            if (i > 0)
035                buf.append(sep);
036            buf.append(newPciAddress());
037        }
038        return buf.toString();
039    }
040
041    @Nonnull
042    public String newPciAddress() {
043        return "0x" + Integer.toHexString(++pciAddress);
044    }
045}
