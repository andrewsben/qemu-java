005package org.anarres.qemu.exec;
006
007import java.util.HashMap;
008import java.util.List;
009import java.util.Map;
010import javax.annotation.Nonnull;
011import org.anarres.qemu.exec.util.QEmuIdAllocator;
012
013/**
014 *
015 * @author shevek
016 */
017public class QEmuDeviceOption extends AbstractQEmuOption {
018
019    public static final String PROP_ID = "id";
020    public static final String PROP_BUS = "bus";
021    public static final String PROP_ADDR = "addr";
022    private final String name;
023    private final Map<String, String> properties = new HashMap<String, String>();
024
025    public QEmuDeviceOption(@Nonnull String name) {
026        this.name = name;
027    }
028
029    @Nonnull
030    public QEmuDeviceOption withProperties(@Nonnull Map<String, String> properties) {
031        this.properties.putAll(properties);
032        return this;
033    }
034
035    @Nonnull
036    public QEmuDeviceOption withProperty(@Nonnull String key, @Nonnull String value) {
037        properties.put(key, value);
038        return this;
039    }
040
041    @Nonnull
042    public QEmuDeviceOption withProperty(@Nonnull String key) {
043        properties.put(key, null);
044        return this;
045    }
046
047    @Nonnull
048    public QEmuDeviceOption withId(@Nonnull String id) {
049        return withProperty(PROP_ID, id);
050    }
051
052    @Override
053    public void appendTo(List<? super String> line) {
054        StringBuilder buf = new StringBuilder(name);
055        appendTo(buf, properties);
056        add(line, "-device", buf);
057    }
058
059    public static class Pci extends QEmuDeviceOption {
060
061        // virtio-balloon-pci,id=balloon0,bus=pci.0,addr=0x6
062        public Pci(@Nonnull String name) {
063            super(name);
064            withBus("pci.0");
065        }
066
067        @Override
068        public Pci withId(@Nonnull String id) {
069            super.withId(id);
070            return this;
071        }
072
073        @Nonnull
074        public Pci withBus(@Nonnull String bus) {
075            withProperty(PROP_BUS, bus);
076            return this;
077        }
078
079        @Nonnull
080        public Pci withAddress(@Nonnull String address) {
081            withProperty(PROP_ADDR, address);
082            return this;
083        }
084
085        @Nonnull
086        public Pci withAddress(@Nonnull QEmuIdAllocator allocator) {
087            withProperty(PROP_ADDR, allocator.newPciAddress());
088            return this;
089        }
090    }
091
092    public static class VirtioBalloon extends Pci {
093
094        public VirtioBalloon() {
095            super("virtio-balloon-pci");
096        }
097    }
098
099    public static class VirtioNet extends Pci {
100
101        // virtio-net-pci,netdev=hostnet0,id=net0,mac=fa:16:3e:13:ff:1f,bus=pci.0,addr=0x3
102        public static final String PROP_NETDEV = "netdev";
103        public static final String PROP_MAC = "mac";
104
105        public VirtioNet() {
106            super("virtio-net-pci");
107        }
108
109        @Nonnull
110        public VirtioNet withMac(String mac) {
111            withProperty(PROP_MAC, mac);
112            return this;
113        }
114    }
115
116    public static class VirtioBlock extends Pci {
117        // virtio-blk-pci,scsi=off,bus=pci.0,addr=0x5,drive=drive-virtio-disk0,id=virtio-disk0,bootindex=1
118
119        public static final String PROP_SCSI = "scsi";
120        public static final String PROP_DRIVE = "drive";
121        public static final String PROP_BOOTINDEX = "bootindex";
122
123        public VirtioBlock() {
124            super("virtio-blk-pci");
125        }
126
127        @Nonnull
128        public VirtioBlock withDrive(String id) {
129            withProperty(PROP_DRIVE, id);
130            return this;
131        }
132
133        @Nonnull
134        public VirtioBlock withDrive(QEmuDriveOption option) {
135            return withDrive(option.id);
136        }
137    }
138
139    public static class Piix3Usb extends Pci {
140
141        // piix3-usb-uhci,id=usb,bus=pci.0,addr=0x1.0x2
142        public Piix3Usb() {
143            super("piix3-usb-uhci");
144            withId("usb");
145        }
146
147        @Override
148        public Piix3Usb withAddress(@Nonnull QEmuIdAllocator allocator) {
149            withProperty(PROP_ADDR, allocator.newPciAddresses(2, "."));
150            return this;
151        }
152    }

161        public Isa withId(@Nonnull String id) {
162            super.withId(id);
163            return this;
164        }
165    }

171
172        public IsaSerial() {
173            super("isa-serial");
174        }
175
176        @Override
177        public IsaSerial withId(@Nonnull String id) {
178            super.withId(id);
179            return this;
180        }
181
182        @Nonnull
183        public IsaSerial withChardev(String id) {
184            withProperty(PROP_CHARDEV, id);
185            return this;
186        }
187    }
