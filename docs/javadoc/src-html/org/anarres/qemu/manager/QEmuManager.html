005package org.anarres.qemu.manager;
006
007import java.io.IOException;
008import java.net.ConnectException;
009import java.net.InetSocketAddress;
010import java.util.HashMap;
011import java.util.List;
012import java.util.Map;
013import java.util.UUID;
014import javax.annotation.Nonnull;
015import org.anarres.qemu.exec.QEmuChardevOption;
016import org.anarres.qemu.exec.QEmuCommandLine;
017import org.anarres.qemu.exec.QEmuIdOption;
018import org.anarres.qemu.exec.QEmuMonitorOption;
019import org.anarres.qemu.exec.host.chardev.TcpCharDevice;
020
021/**
022 *
023 * @author shevek
024 */
025public class QEmuManager {
026
027    private final Map<UUID, QEmuProcess> processes = new HashMap<UUID, QEmuProcess>();
028
029    @Nonnull
030    public QEmuProcess execute(QEmuCommandLine commandLine) throws IOException, InterruptedException {
031        // QEmuIdOption idOption = commandLine.getOption(QEmuIdOption.class);
032        UUID uuid;
033        UUID:
034        {
035            QEmuIdOption idOption = commandLine.getOption(QEmuIdOption.class);
036            if (idOption == null)
037                break UUID;
038            uuid = idOption.getUuid();
039
040        }
041        InetSocketAddress qmpAddress = null;
042        QMP:
043        {
044            QEmuMonitorOption monitorOption = commandLine.getOption(QEmuMonitorOption.class);
045            if (monitorOption == null)
046                break QMP;
047            String monitorChardev = monitorOption.chardev;
048            for (QEmuChardevOption chardevOption : commandLine.getOptions(QEmuChardevOption.class)) {
049                if (chardevOption.id.equals(monitorChardev)) {
050                    if (chardevOption.device instanceof TcpCharDevice) {
051                        TcpCharDevice tcpDevice = (TcpCharDevice) chardevOption.device;
052                        qmpAddress = tcpDevice.getAddress();
053                    }
054                }
055            }
056        }
057        List<String> commandWords = commandLine.toCommandWords();
058        ProcessBuilder builder = new ProcessBuilder(commandWords);
059        Process process = builder.start();
060        QEmuProcess qEmuProcess = new QEmuProcess(process, qmpAddress);
061        CONNECT:
062        {
063            for (int i = 0; i < 20; i++) {
064                try {
065                    qEmuProcess.getConnection();
066                    break CONNECT;
067                } catch (ConnectException e) {
068                    // Process not started yet.
069                    // System.err.println(e);
070                }
071                Thread.sleep(200);
072            }
073        }
074        return qEmuProcess;
075    }
076}
