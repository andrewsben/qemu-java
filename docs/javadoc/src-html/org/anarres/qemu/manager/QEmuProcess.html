005package org.anarres.qemu.manager;
006
007import java.io.IOException;
008import java.net.InetSocketAddress;
009import javax.annotation.CheckForNull;
010import javax.annotation.Nonnull;
011import org.anarres.qemu.qapi.api.QuitCommand;
012import org.anarres.qemu.qapi.common.QApiConnection;
013import org.anarres.qemu.qapi.common.QApiException;
014
015/**
016 *
017 * @author shevek
018 */
019public class QEmuProcess {
020
021    private final Process process;
022    private final InetSocketAddress monitor;
023    private QApiConnection connection;
024    private final Object lock = new Object();
025    private final IOBuffer stdout = new IOBuffer();
026    private final IOBuffer stderr = new IOBuffer();
027
028    public QEmuProcess(@Nonnull Process process, @CheckForNull InetSocketAddress monitor) {
029        this.process = process;
030        this.monitor = monitor;
031
032        new IOThread(process.getInputStream(), stdout).start();
033        new IOThread(process.getErrorStream(), stderr).start();
034    }
035
036    @CheckForNull
037    public QApiConnection getConnection() throws IOException {
038        synchronized (lock) {
039            if (monitor == null)
040                return null;
041            if (connection != null)
042                return connection;
043            connection = new QApiConnection(monitor);
044            return connection;
045        }
046    }
047
048    public void destroy() throws IOException, QApiException {
049        try {
050            getConnection().call(new QuitCommand());
051        } catch (IOException e) {
052        }
053        process.destroy();
054    }
055}
